#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Keypad.h>
#include <Stepper.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);

#define STEPS 2048
Stepper stepper_motor(STEPS, 10, 12, 11, 13);

String password = "5809";
String input_password;

const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};
byte rowPins[ROWS] = {9, 8, 7, 6};
byte colPins[COLS] = {5, 4, 3, 2};
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

#define Buzzer 4

void setup() {
  pinMode(Buzzer, OUTPUT);
  stepper_motor.setSpeed(10);
  lcd.init(); lcd.backlight();
  lcd.setCursor(0,0); lcd.print("PASSWORD LOCK");
  delay(2000); lcd.clear();
}

void loop() {
  lcd.setCursor(0, 0);
  lcd.print("Enter Password:");
  input_password = "";
  while (true) {
    char key = keypad.getKey();
    if (key) {
      if (key == '#') {
        verifyPassword();
        break;
      } else if (key == '*') {
        input_password = "";
        lcd.setCursor(0, 1); lcd.print("                ");
      } else {
        input_password += key;
        lcd.setCursor(0, 1); lcd.print(input_password);
      }
    }
  }
}

void verifyPassword() {
  lcd.clear();
  if (input_password == password) {
    lcd.print("Access Granted");
    unlockDoor();
  } else {
    lcd.print("Access Denied");
    wrongAlert();
  }
  delay(2000); lcd.clear();
}

void unlockDoor() {
  stepper_motor.step(1024);
  delay(2000);
  stepper_motor.step(-1024);
}

void wrongAlert() {
  for (int i=0; i<3; i++) {
    digitalWrite(Buzzer, HIGH);
    delay(200);
    digitalWrite(Buzzer, LOW);
    delay(200);
  }
}
